### 1.2 核心关系

| 关系 | 说明 | 类比 |
|-----|------|------|
| **xopc.ai → xopc-idealab** | 品牌 → 产品 | 苹果 → iPhone |
| **xopc-idealab → xopcbot** | 产品 → 框架 | 微信 → iOS |
| **xopcbot → AI 模型** | 框架 → 服务 | iOS → iCloud |

**关键原则**：
- xopc-idealab 是**独立产品**，不是 xopcbot 的附属
- xopcbot 是**可替换框架**，未来可以迁移到其他框架
- xopc.ai 的护城河在**产品层**（催化逻辑 + 品味模型），不在框架层

## 四、接口定义

### 4.1 xopc-idealab → xopcbot（插件接口）

```typescript
// idealab 作为 xopcbot 插件，需要实现以下接口

interface CatalystPlugin {
  // 插件元数据
  name: 'idealab';
  version: '1.0.0';
  
  // 初始化（xopcbot 启动时调用）
  init(ctx: XopcbotContext): Promise<void>;
  
  // 消息处理钩子（收到用户消息时调用）
  onMessage?(msg: Message): Promise<HookResult>;
  
  // 心跳钩子（Cron 触发时调用）
  onHeartbeat?(): Promise<HookResult>;
  
  // 反馈处理钩子（用户回复反馈时调用）
  onFeedback?(feedback: Feedback): Promise<HookResult>;
}

// xopcbot 提供给插件的能力
interface XopcbotContext {
  // AI 调用
  ai: {
    generate(prompt: string, options?: AIOptions): Promise<string>;
    chat(messages: Message[], options?: AIOptions): Promise<string>;
  };
  
  // 通道发送
  channel: {
    send(userId: string, message: string): Promise<void>;
    reply(messageId: string, content: string): Promise<void>;
  };
  
  // 存储
  storage: {
    get(key: string): Promise<any>;
    set(key: string, value: any): Promise<void>;
    query(sql: string, params?: any[]): Promise<any[]>;
  };
  
  // Cron 调度
  cron: {
    schedule(expr: string, handler: () => Promise<void>): Promise<void>;
    cancel(jobId: string): Promise<void>;
  };
}
```

### 4.2 核心领域模型

```typescript
// Idea 实体
interface Idea {
  id: string;                    // UUID
  userId: string;                // 用户 ID
  content: string;               // 原始想法
  inputType: 'text' | 'voice' | 'link' | 'image';
  
  // 阶段管理
  stage: IdeaStage;              // fresh/incubating/deepening/dormant
  passionScore: number;          // 热情度 1-10
  
  // 心跳管理
  lastBeatAt: number;            // 上次心跳时间
  nextBeatAt: number;            // 下次心跳时间
  beatCount: number;             // 心跳次数
  
  // 催化记录
  catalysis: CatalysisRecord[];  // 催化历史
  
  // 元数据
  tags: string[];
  createdAt: number;
  updatedAt: number;
}

// 阶段定义
type IdeaStage = 
  | 'fresh'      // 新鲜期 (0-24h)
  | 'incubating' // 孵化期 (24h-7d)
  | 'deepening'  // 深化期 (7d-30d)
  | 'dormant';   // 休眠期 (30d+)

// 催化记录
interface CatalysisRecord {
  id: string;
  ideaId: string;
  stage: IdeaStage;              // 催化时的阶段
  questions: string[];           // AI 生成的问题
  suggestions: string[];         // 行动建议
  confidence: number;            // AI 信心分数 0-1
  
  // 用户反馈
  feedback?: 'helpful' | 'not_helpful' | 'ignored';
  responseTime?: number;         // 用户回应时间（秒）
  responseLength?: number;       // 用户回应长度
  
  createdAt: number;
}

// 用户画像
interface UserProfile {
  userId: string;
  
  // 品味偏好
  questionStyle: 'direct' | 'gentle' | 'challenging';
  focusAreas: Array<'product' | 'tech' | 'market' | 'design'>;
  actionOrientation: 'high' | 'medium' | 'low';
  
  // 行为统计
  avgResponseTime: number;       // 平均回应时间（秒）
  avgResponseLength: number;     // 平均回应长度（字）
  helpfulRate: number;           // 有用反馈比例
  bestCatalystTime: 'morning' | 'afternoon' | 'night';
  
  // 催化节奏偏好
  preferredInterval: {
    incubating: number;          // 孵化期间隔（小时）
    deepening: number;           // 深化期间隔（小时）
    dormant: number;             // 休眠期间隔（小时）
  };
  
  updatedAt: number;
}
```

interface CatalystModule {

  new EvaluateModule(),      // E - 评估
  new ActModule(),           // A - 行动
  new ExternalSignalsModule(), // 外部信号（V2）
  new ReportGeneratorModule(), // 报告生成
];
